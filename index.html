public int compress(char[] chars) {
    int n = chars.length;
    if (n == 1) return 1;

    int writeIndex = 0; // 寫入位置
    int count = 1; // 當前字符的計數

    for (int i = 1; i <= n; i++) {
        if (i == n || chars[i] != chars[i - 1]) {
            chars[writeIndex++] = chars[i - 1];
            if (count > 1) {
                for (char c : Integer.toString(count).toCharArray()) {
                    chars[writeIndex++] = c;
                }
            }
            count = 1;
        } else {
            count++;
        }
    }

    return writeIndex;
}
